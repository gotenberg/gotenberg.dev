import ApiEndpoint from "@site/src/components/documentation/ApiEndpoint";
import SyntaxWatermarkPDFEngines from "./_syntax_watermark_pdfengines.mdx";

Adds a watermark _behind_ the content of each page during post-processing. Sources: `text`, `image`, or `pdf`.

<SyntaxWatermarkPDFEngines />

{props.showCurlExample && (

<ApiEndpoint
  formFields={[
    {
      name: "watermarkSource",
      type: "enum",
      required: false,
      description:
        "The watermark source type. Options: 'text', 'image', 'pdf'.",
    },
    {
      name: "watermarkExpression",
      type: "string",
      required: false,
      description:
        "The watermark content. For 'text', the string to render. For 'image' or 'pdf', the filename of the uploaded watermark file.",
    },
    {
      name: "watermarkPages",
      type: "string",
      required: false,
      description:
        "Page ranges to watermark (e.g., '1-3', '5'). Empty means all pages.",
    },
    {
      name: "watermarkOptions",
      type: "json",
      required: false,
      description:
        "Advanced options in JSON format (e.g., font, color, rotation, opacity, scaling).",
    },
  ]}
  formFiles={[
    {
      name: "watermark",
      type: "file",
      required: false,
      description:
        "An image or PDF file used as watermark source (required when watermarkSource is 'image' or 'pdf').",
    },
  ]}
  curl={`
curl \\
--request ${props.method} http://localhost:3000${props.path} \\
${props.curlFormData}
--form watermarkSource=text \\
--form watermarkExpression=CONFIDENTIAL \\
--form 'watermarkOptions={"opacity":0.25,"rotation":45}' \\
${props.curlOutput}
`}
/>
)}
