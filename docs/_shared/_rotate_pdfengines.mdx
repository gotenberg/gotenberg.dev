import ApiEndpoint from "@site/src/components/documentation/ApiEndpoint";

Rotates pages by `90`, `180`, or `270` degrees during post-processing.

{props.showCurlExample && (

<ApiEndpoint
  formFields={[
    {
      name: "rotateAngle",
      type: "enum",
      required: false,
      description: "The rotation angle. Options: '90', '180', '270'.",
    },
    {
      name: "rotatePages",
      type: "string",
      required: false,
      description:
        "Page ranges to rotate (e.g., '1-3', '5'). Empty means all pages.",
    },
  ]}
  curl={`
curl \\
--request ${props.method} http://localhost:3000${props.path} \\
${props.curlFormData}
--form rotateAngle=90 \\
${props.curlOutput}
`}
/>
)}
