import ApiEndpoint from "@site/src/components/documentation/ApiEndpoint";

Set passwords to control PDF access. The **user password** is required to open the PDF; the **owner password** controls permissions (printing, copying, editing).

Encryption strength (e.g., AES-256) depends on the active PDF Engine. See [PDF Engines Configuration](/docs/configuration#pdf-engines).

{props.showFormFields && props.showCurlExample && (

<ApiEndpoint
  formFields={[
    {
      name: "userPassword",
      type: "string",
      defaultValue: "None",
      description: "The password required to open the PDF.",
    },
    {
      name: "ownerPassword",
      type: "string",
      defaultValue: "None",
      description:
        "The password required to change permissions or edit the PDF.",
    },
  ]}
  curl={`
curl \\
--request ${props.method} http://localhost:3000${props.path} \\
${props.curlFormData}
--form userPassword=my_user_password \\
--form ownerPassword=my_owner_password \\
${props.curlOutput}
`}
/>
)}

{!props.showFormFields && props.showCurlExample && (

<ApiEndpoint
  curl={`
curl \\
--request ${props.method} http://localhost:3000${props.path} \\
${props.curlFormData}
--form userPassword=my_user_password \\
--form ownerPassword=my_owner_password \\
${props.curlOutput}
`}
/>
)}
