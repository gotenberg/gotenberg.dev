import ApiEndpoint from "@site/src/components/documentation/ApiEndpoint";

Inject XMP metadata (Author, Title, Copyright, Keywords, etc.) into the PDF as a JSON object.

Not all tags are writable. Gotenberg uses ExifTool under the hood. See the [XMP Tag Name documentation](https://exiftool.org/TagNames/XMP.html#pdf) for valid keys. Writing metadata usually breaks **PDF/A** compliance.

{(props.showFormFields || props.showCurlExample) && (

<ApiEndpoint
  formFields={
    props.showFormFields
      ? [
          {
            name: "metadata",
            type: "json",
            defaultValue: "None",
            description: "Writes metadata (Author, Title, etc.).",
          },
        ]
      : undefined
  }
  curl={
    props.showCurlExample
      ? `
curl \\
--request ${props.method} http://localhost:3000${props.path} \\
${props.curlFormData}
--form 'metadata={"Author":"Julien Neuhart","Copyright":"Julien Neuhart","CreationDate":"2006-09-18T16:27:50-04:00","Creator":"Gotenberg","Keywords":["first","second"],"Marked":true,"ModDate":"2006-09-18T16:27:50-04:00","PDFVersion":1.7,"Producer":"Gotenberg","Subject":"Sample","Title":"Sample","Trapped":"Unknown"}' \\
${props.curlOutput}
`
      : undefined
  }
/>
)}
