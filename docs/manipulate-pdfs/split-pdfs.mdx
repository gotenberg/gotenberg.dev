Splits one or more PDF files by extracting specific pages or splitting into intervals. Returns a ZIP archive of individual files or a unified PDF.

<SyntaxValidationSplitPDFEngines />

## Basics

<ApiEndpoint
  method="POST"
  path="/forms/pdfengines/split"
  headers={[
    {
      name: "Gotenberg-Output-Filename",
      type: "string",
      required: false,
      description:
        "The filename of the resulting file - Gotenberg automatically appends the file extension. Defaults to a random UUID filename.",
    },
    {
      name: "Gotenberg-Trace",
      type: "string",
      required: false,
      description:
        "A custom request ID to identify the request in the logs; overrides the default UUID.",
    },
  ]}
  formFields={[
    {
      name: "splitMode",
      type: "enum",
      required: true,
      description:
        "Activates the splitting engine. Options: 'intervals' or 'pages'.",
    },
    {
      name: "splitSpan",
      type: "string",
      required: true,
      description:
        "The rule for splitting. If mode is 'intervals', defines the chunk size (e.g. '2'). If mode is 'pages', defines the page ranges.",
    },
    {
      name: "splitUnify",
      type: "boolean",
      defaultValue: "false",
      description:
        "Only for 'pages' mode. If true, puts all extracted pages from a PDF into a single PDF file. If false, creates a separate file for each range.",
    },
  ]}
  formFiles={[
    {
      name: "files",
      type: "file[]",
      required: true,
      description: "PDF files to split.",
    },
  ]}
  curl={`
curl \\
--request POST http://localhost:3000/forms/pdfengines/split \\
--form files=@/path/to/file.pdf \\
--form splitMode=intervals \\
--form splitSpan=1 \\
-o my.zip`}
  responses={[
    {
      status: 200,
      message: "OK",
      description: "The split PDF files (ZIP) or unified PDF.",
      body: `
Content-Disposition: attachment; filename={output-filename.ext}
Content-Type: {content-type}
Content-Length: {content-length}
Gotenberg-Trace: {trace}
Body: {output-file}
      `,
    },
    {
      status: 400,
      message: "Bad Request",
      description: "The request had one ore more invalid form fields.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: {error}
      `,
    },
    {
      status: 503,
      message: "Service Unavailable",
      description:
        "The request did not complete within the configured maximum duration.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: Service Unavailable
      `,
    },
  ]}
/>
