Adds a watermark _behind_ the content of each page of one or more PDF files. Use `text` for a simple label, `image` to tile an uploaded image, or `pdf` to place a full-page PDF underneath.

<SyntaxWatermarkPDFEngines />

<ApiEndpoint
  method="POST"
  path="/forms/pdfengines/watermark"
  headers={[
    {
      name: "Gotenberg-Output-Filename",
      type: "string",
      required: false,
      description:
        "The filename of the resulting file - Gotenberg automatically appends the file extension. Defaults to a random UUID filename.",
    },
    {
      name: "Gotenberg-Trace",
      type: "string",
      required: false,
      description:
        "A custom request ID to identify the request in the logs; overrides the default UUID.",
    },
  ]}
  formFields={[
    {
      name: "watermarkSource",
      type: "enum",
      required: true,
      description:
        "The watermark source type. Options: 'text', 'image', 'pdf'.",
    },
    {
      name: "watermarkExpression",
      type: "string",
      required: true,
      description:
        "The watermark content. For 'text', the string to render. For 'image' or 'pdf', the filename of the uploaded watermark file.",
    },
    {
      name: "watermarkPages",
      type: "string",
      required: false,
      description:
        "Page ranges to watermark (e.g., '1-3', '5'). Empty means all pages.",
    },
    {
      name: "watermarkOptions",
      type: "json",
      required: false,
      description:
        "Advanced options in JSON format (e.g., font, color, rotation, opacity, scaling).",
    },
  ]}
  formFiles={[
    {
      name: "files",
      type: "file[]",
      required: true,
      description: "PDF files to watermark.",
    },
    {
      name: "watermark",
      type: "file",
      required: false,
      description:
        "An image or PDF file used as watermark source (required when watermarkSource is 'image' or 'pdf').",
    },
  ]}
  curl={`
curl \\
--request POST http://localhost:3000/forms/pdfengines/watermark \\
--form files=@/path/to/file.pdf \\
--form watermarkSource=text \\
--form watermarkExpression=CONFIDENTIAL \\
--form 'watermarkOptions={"opacity":0.25,"rotation":45}' \\
-o my.pdf`}
  responses={[
    {
      status: 200,
      message: "OK",
      description: "The watermarked PDF file.",
      body: `
Content-Disposition: attachment; filename={output-filename.pdf}
Content-Type: {content-type}
Content-Length: {content-length}
Gotenberg-Trace: {trace}
Body: {output-file}
      `,
    },
    {
      status: 400,
      message: "Bad Request",
      description: "The request had one ore more invalid form fields.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: {error}
      `,
    },
    {
      status: 503,
      message: "Service Unavailable",
      description:
        "The request did not complete within the configured maximum duration.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: Service Unavailable
      `,
    },
  ]}
/>
