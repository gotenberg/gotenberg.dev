Converts one or more PDF files into PDF/A (archival) or PDF/UA (accessibility) format.

:::info LibreOffice Dependency

This endpoint relies on **LibreOffice** to re-process documents. This operation is more resource-intensive (CPU & RAM) than simple merges or splits.

:::

<ApiEndpoint
  method="POST"
  path="/forms/pdfengines/convert"
  headers={[
    {
      name: "Gotenberg-Output-Filename",
      type: "string",
      required: false,
      description:
        "The filename of the resulting file - Gotenberg automatically appends the file extension. Defaults to a random UUID filename.",
    },
    {
      name: "Gotenberg-Trace",
      type: "string",
      required: false,
      description:
        "A custom request ID to identify the request in the logs; overrides the default UUID.",
    },
  ]}
  formFiles={[
    {
      name: "files",
      type: "file[]",
      required: true,
      description: "PDF files to convert.",
    },
  ]}
  formFields={[
    {
      name: "pdfa",
      type: "enum",
      required: false,
      description:
        "Converts to a specific PDF/A archival standard. Options: 'PDF/A-1b', 'PDF/A-2b', 'PDF/A-3b'.",
    },
    {
      name: "pdfua",
      type: "boolean",
      defaultValue: "false",
      description: "Enables PDF/UA (Universal Accessibility) compliance.",
    },
  ]}
  curl={`
curl \\
--request POST http://localhost:3000/forms/pdfengines/convert \\
--form files=@/path/to/pdf.pdf \\
--form pdfa=PDF/A-1b \\
--form pdfua=true \\
-o my.pdf`}
  responses={[
    {
      status: 200,
      message: "OK",
      description: "The converted PDF file.",
      body: `
Content-Disposition: attachment; filename={output-filename.pdf}
Content-Type: {content-type}
Content-Length: {content-length}
Gotenberg-Trace: {trace}
Body: {output-file}
      `,
    },
    {
      status: 400,
      message: "Bad Request",
      description: "The request had one ore more invalid form fields.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: {error}
      `,
    },
    {
      status: 503,
      message: "Service Unavailable",
      description:
        "The request did not complete within the configured maximum duration.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: Service Unavailable
      `,
    },
  ]}
/>
