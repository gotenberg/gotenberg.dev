title: Rotate PDFs

import ApiEndpoint from "@site/src/components/documentation/ApiEndpoint";
import Sponsors from "@site/src/components/documentation/Sponsors";

Rotates pages of one or more PDF files by a given angle. You can target all pages or specific page ranges.

<ApiEndpoint
  method="POST"
  path="/forms/pdfengines/rotate"
  headers={[
    {
      name: "Gotenberg-Output-Filename",
      type: "string",
      required: false,
      description:
        "The filename of the resulting file - Gotenberg automatically appends the file extension. Defaults to a random UUID filename.",
    },
    {
      name: "Gotenberg-Trace",
      type: "string",
      required: false,
      description:
        "A custom request ID to identify the request in the logs; overrides the default UUID.",
    },
  ]}
  formFields={[
    {
      name: "rotateAngle",
      type: "enum",
      required: true,
      description: "The rotation angle. Options: '90', '180', '270'.",
    },
    {
      name: "rotatePages",
      type: "string",
      required: false,
      description:
        "Page ranges to rotate (e.g., '1-3', '5'). Empty means all pages.",
    },
  ]}
  formFiles={[
    {
      name: "files",
      type: "file[]",
      required: true,
      description: "PDF files to rotate.",
    },
  ]}
  curl={`
curl \\
--request POST http://localhost:3000/forms/pdfengines/rotate \\
--form files=@/path/to/file.pdf \\
--form rotateAngle=90 \\
-o my.pdf`}
  responses={[
    {
      status: 200,
      message: "OK",
      description: "The rotated PDF file.",
      body: `
Content-Disposition: attachment; filename={output-filename.pdf}
Content-Type: {content-type}
Content-Length: {content-length}
Gotenberg-Trace: {trace}
Body: {output-file}
      `,
    },
    {
      status: 400,
      message: "Bad Request",
      description: "The request had one ore more invalid form fields.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: {error}
      `,
    },
    {
      status: 503,
      message: "Service Unavailable",
      description:
        "The request did not complete within the configured maximum duration.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: Service Unavailable
      `,
    },
  ]}
/>
