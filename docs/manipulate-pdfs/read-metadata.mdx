Extracts metadata (Exif, XMP, PDF properties, etc.) from one or more PDF files using **ExifTool**. Returns a JSON object keyed by filename.

:::info ExifTool Tags
The returned keys correspond to ExifTool tag names.
[See the ExifTool Tag Names documentation](https://exiftool.org/TagNames/PDF.html) for reference on what fields might be returned.
:::

<ApiEndpoint
  method="POST"
  path="/forms/pdfengines/metadata/read"
  headers={[
    {
      name: "Gotenberg-Trace",
      type: "string",
      required: false,
      description:
        "A custom request ID to identify the request in the logs; overrides the default UUID.",
    },
  ]}
  formFiles={[
    {
      name: "files",
      type: "file[]",
      required: true,
      description: "PDF files to analyze.",
    },
  ]}
  curl={`
curl \\
--request POST http://localhost:3000/forms/pdfengines/metadata/read \\
--form files=@/path/to/invoice.pdf \\
--form files=@/path/to/report.pdf`}
  responses={[
    {
      status: 200,
      message: "OK",
      description: "The metadata in JSON format.",
      body: `

      "Title": "Invoice #001",
      "CreateDate": "2024:03:05 09:15:32Z",
      "PageCount": 2,
      "Producer": "Gotenberg",
      "MIMEType": "application/pdf"
    },
    "report.pdf": {
      "PDFVersion": 1.4,
      "Author": "John Doe",
      "PageCount": 15
      // ...
    }
}
      `,
    },
    {
      status: 400,
      message: "Bad Request",

    },
  ]}
/>
