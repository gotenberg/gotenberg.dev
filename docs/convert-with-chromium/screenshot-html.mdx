Captures a screenshot of an `index.html` file (and optional assets) using Headless Chromium. Reference additional CSS, images, and fonts in your HTML using relative paths.

<ConfigurationInfo />

## Basics

<ApiEndpoint
  method="POST"
  path="/forms/chromium/screenshot/html"
  headers={[
    {
      name: "Gotenberg-Output-Filename",
      type: "string",
      required: false,
      description:
        "The filename of the resulting file - Gotenberg automatically appends the file extension. Defaults to a random UUID filename.",
    },
    {
      name: "Gotenberg-Trace",
      type: "string",
      required: false,
      description:
        "A custom request ID to identify the request in the logs; overrides the default UUID.",
    },
  ]}
  formFiles={[
    {
      name: "index.html",
      type: "file",
      required: true,
      description: "An HTML file named 'index.html' to convert into an image.",
    },
  ]}
  curl={`
curl \\
--request POST http://localhost:3000/forms/chromium/screenshot/html \\
--form files=@/path/to/index.html \\
-o my.jpeg`}
  responses={[
    {
      status: 200,
      message: "OK",
      description: "The screenshot has been successfully created.",
      body: `
Content-Disposition: attachment; filename={output-filename.ext}
Content-Type: {content-type}
Content-Length: {content-length}
Gotenberg-Trace: {trace}
Body: {output-file}
      `,
    },
    {
      status: 400,
      message: "Bad Request",
      description: "The request had one ore more invalid form fields.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: {error}
      `,
    },
    {
      status: 503,
      message: "Service Unavailable",
      description:
        "The request did not complete within the configured maximum duration.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: Service Unavailable
      `,
    },
  ]}
/>
