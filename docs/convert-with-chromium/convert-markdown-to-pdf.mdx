Converts Markdown content (with [MathJaX](https://www.mathjax.org/) support) to PDF using Headless Chromium. You provide an `index.html` template and one or more `.md` files. Gotenberg converts the Markdown to HTML and injects it into your template.

:::info Template Configuration

The `index.html` file **must** include the following Go template action where the content should appear:

```html {8}
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="utf-8" />
    <title>My PDF</title>
  </head>
  <body>
    {{ toHTML "file.md" }}
  </body>
</html>
```

:::

## Basics

<ApiEndpoint
  method="POST"
  path="/forms/chromium/convert/markdown"
  headers={[
    {
      name: "Gotenberg-Output-Filename",
      type: "string",
      required: false,
      description:
        "The filename of the resulting file - Gotenberg automatically appends the file extension. Defaults to a random UUID filename.",
    },
    {
      name: "Gotenberg-Trace",
      type: "string",
      required: false,
      description:
        "A custom request ID to identify the request in the logs; overrides the default UUID.",
    },
  ]}
  formFiles={[
    {
      name: "index.html",
      type: "file",
      required: true,
      description:
        "An HTML file named 'index.html' to wrap the markdown content.",
    },
    {
      name: "*.md",
      type: "file[]",
      required: true,
      description: "At least one markdown file.",
    },
  ]}
  curl={`
curl \\
--request POST http://localhost:3000/forms/chromium/convert/markdown \\
--form files=@/path/to/index.html \\
--form files=@/path/to/file.md \\
-o my.pdf`}
  responses={[
    {
      status: 200,
      message: "OK",
      description: "The PDF file has been successfully created.",
      body: `
Content-Disposition: attachment; filename={output-filename.pdf}
Content-Type: {content-type}
Content-Length: {content-length}
Gotenberg-Trace: {trace}
Body: {output-file}
      `,
    },
    {
      status: 400,
      message: "Bad Request",
      description: "The request had one ore more invalid form fields.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: {error}
      `,
    },
    {
      status: 503,
      message: "Service Unavailable",
      description:
        "The request did not complete within the configured maximum duration.",
      body: `
Content-Type: text/plain; charset=UTF-8
Gotenberg-Trace: {trace}
Body: Service Unavailable
      `,
    },
  ]}
/>
